JavaScriptの3D表示例（Three.js）

Three.jsを使用して3Dオブジェクトを描画する方法を説明します。

Three.jsを使って3Dシーンを作成し、立方体を描画します。
技術: シーンの作成、カメラの設定、レンダラーの設定、立方体の作成、アニメーションの実装
// シーンの作成
const scene = new THREE.Scene();

// カメラの設定
// THREE.PerspectiveCamera(fov, aspect, near, far)
// fov: 視野角, aspect: アスペクト比, near: カメラからの最短距離, far: カメラからの最長距離
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
camera.position.z = 5;

// レンダラーの設定
const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('3dCanvas') });
renderer.setSize(window.innerWidth * 0.8, 300);

// 立方体の作成
const geometry = new THREE.BoxGeometry();
const material = new THREE.MeshBasicMaterial({ color: 0x00ff00 });
const cube = new THREE.Mesh(geometry, material);
scene.add(cube);

// エッジの作成
const edges = new THREE.EdgesGeometry(geometry);
const lineMaterial = new THREE.LineBasicMaterial({ color: 0x000000 });
const line = new THREE.LineSegments(edges, lineMaterial);
scene.add(line);

// アニメーションの実装
function animate() {
requestAnimationFrame(animate);
cube.rotation.x += 0.01; // 立方体をx軸で回転
cube.rotation.y += 0.01; // 立方体をy軸で回転
line.rotation.x += 0.01; // エッジをx軸で回転
line.rotation.y += 0.01; // エッジをy軸で回転
renderer.render(scene, camera); // シーンとカメラをレンダリング
}
animate();
